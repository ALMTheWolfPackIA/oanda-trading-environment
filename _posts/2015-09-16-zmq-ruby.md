Example subscriber in **Ruby** 

{% highlight js %}
require 'rubygems'
require 'ffi-rzmq'

context = ZMQ::Context.new()

subscriber = context.socket(ZMQ::SUB)
subscriber.connect("tcp://172.16.95.158:5550")
filter = ''
subscriber.setsockopt(ZMQ::SUBSCRIBE, filter)

s = ''
loop do
  subscriber.recv_string(s)
  puts s
end
{% endhighlight %}
